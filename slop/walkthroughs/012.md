Add after `post_process_yolo()`:

```rust
fn non_max_suppression(&self, mut boxes: Vec<BBox>, iou_threshold: f32) -> Vec<BBox> {
    // 1. Sort by confidence (highest first)
    boxes.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    let mut keep = Vec::new();
    let mut suppressed = vec![false; boxes.len()];

    for i in 0..boxes.len() {
        if suppressed[i] {
            continue;
        }
        keep.push(boxes[i].clone());

        // Suppress overlapping boxes of the same class
        for j in (i + 1)..boxes.len() {
            if suppressed[j] {
                continue;
            }

            // Only compare same-class boxes
            if boxes[i].class_id != boxes[j].class_id {
                continue;
            }

            let iou = self.calculate_iou(&boxes[i], &boxes[j]);
            if iou > iou_threshold {
                suppressed[j] = true;
            }
        }
    }

    keep
}

fn calculate_iou(&self, box1: &BBox, box2: &BBox) -> f32 {
    // Intersection coordinates
    let x1 = box1.x1.max(box2.x1);
    let y1 = box1.y1.max(box2.y1);
    let x2 = box1.x2.min(box2.x2);
    let y2 = box1.y2.min(box2.y2);

    // No overlap
    if x2 < x1 || y2 < y1 {
        return 0.0;
    }

    let intersection = (x2 - x1) * (y2 - y1);
    let area1 = (box1.x2 - box1.x1) * (box1.y2 - box1.y1);
    let area2 = (box2.x2 - box2.x1) * (box2.y2 - box2.y1);
    let union = area1 + area2 - intersection;

    intersection / union
}
```

fn count_by_class(bboxes: &[crate::models::layout::BBox]) -> std::collections::HashMap<String, usize> {
    use std::collections::HashMap;

    let mut counts = HashMap::new();
    for bbox in bboxes {
        *counts.entry(bbox.class_name.clone()).or_insert(0) += 1;
    }
    counts
}

### Stage 3: Reading Order Sorting (45 min)

**Why:** Documents must flow naturally (top-to-bottom, left-to-right).

File: `pdf-mash/src/models/layout.rs`

Add sorting method:

```rust
pub fn sort_by_reading_order(bboxes: &mut [BBox]) {
    // Multi-column detection threshold
    let column_gap_threshold = 50.0;

    // Primary sort: Y-coordinate (top to bottom)
    // Secondary sort: X-coordinate (left to right)
    bboxes.sort_by(|a, b| {
        let y_diff = (a.y1 - b.y1).abs();

        if y_diff < column_gap_threshold {
            // Same row, sort by X
            a.x1.partial_cmp(&b.x1).unwrap()
        } else {
            // Different rows, sort by Y
            a.y1.partial_cmp(&b.y1).unwrap()
        }
    });
}
```

use crate::models::layout::BBox;
use anyhow::Result;
use image::{DynamicImage, Rgb, RgbImage};
use imageproc::drawing::{draw_hollow_rect_mut, draw_text_mut};
use imageproc::rect::Rect;
use rusttype::{Font, Scale};

pub struct BBoxVisualizer {
    font: Font<'static>,
}

impl BBoxVisualizer {
    pub fn new() -> Result<Self> {
        // Use embedded font for portability
        let font_data = include_bytes!("../../assets/DejaVuSans.ttf");
        let font = Font::try_from_bytes(font_data as &[u8])
            .ok_or_else(|| anyhow::anyhow!("Failed to load font"))?;

        Ok(Self { font })
    }

    pub fn draw_bboxes(&self, image: &DynamicImage, bboxes: &[BBox]) -> Result<RgbImage> {
        let mut annotated = image.to_rgb8();

        for (idx, bbox) in bboxes.iter().enumerate() {
            let color = self.class_color(&bbox.class_name);

            // Draw rectangle
            let rect = Rect::at(bbox.x1 as i32, bbox.y1 as i32)
                .of_size(
                    (bbox.x2 - bbox.x1) as u32,
                    (bbox.y2 - bbox.y1) as u32,
                );
            draw_hollow_rect_mut(&mut annotated, rect, color);

            // Draw label
            let label = format!("{} {:.2}", bbox.class_name, bbox.confidence);
            let scale = Scale::uniform(16.0);

            draw_text_mut(
                &mut annotated,
                color,
                bbox.x1 as i32,
                (bbox.y1 - 20.0).max(0.0) as i32,
                scale,
                &self.font,
                &label,
            );
        }

        Ok(annotated)
    }

    fn class_color(&self, class_name: &str) -> Rgb<u8> {
        match class_name {
            "text" => Rgb([0, 255, 0]),      // Green
            "title" => Rgb([255, 0, 0]),     // Red
            "figure" => Rgb([0, 0, 255]),    // Blue
            "table" => Rgb([255, 255, 0]),   // Yellow
            "equation" => Rgb([255, 0, 255]), // Magenta
            _ => Rgb([128, 128, 128]),       // Gray
        }
    }
}

### Stage 5: Basic Markdown Generation (45 min)

**Why:** Makes detections usable as structured text.

```rust
use crate::models::layout::BBox;
use anyhow::Result;

pub struct MarkdownGenerator {
    include_metadata: bool,
}

impl MarkdownGenerator {
    pub fn new() -> Self {
        Self {
            include_metadata: true,
        }
    }

    pub fn generate(&self, bboxes: &[BBox], _page_text: &str) -> Result<String> {
        let mut md = String::new();

        // Group consecutive text blocks
        let mut current_paragraph = String::new();

        for bbox in bboxes {
            match bbox.class_name.as_str() {
                "title" => {
                    // Flush current paragraph
                    if !current_paragraph.is_empty() {
                        md.push_str(&current_paragraph);
                        md.push_str("\n\n");
                        current_paragraph.clear();
                    }

                    // Add title (we don't have OCR yet, so use placeholder)
                    md.push_str("## [Title]\n\n");

                    if self.include_metadata {
                        md.push_str(&format!(
                            "<!-- bbox: [{:.1}, {:.1}, {:.1}, {:.1}], conf: {:.2} -->\n\n",
                            bbox.x1, bbox.y1, bbox.x2, bbox.y2, bbox.confidence
                        ));
                    }
                }
                "text" => {
                    // Placeholder for text (need OCR)
                    current_paragraph.push_str("[Text block]");
                    current_paragraph.push(' ');
                }
                "figure" => {
                    if !current_paragraph.is_empty() {
                        md.push_str(&current_paragraph);
                        md.push_str("\n\n");
                        current_paragraph.clear();
                    }

                    md.push_str("![Figure](figure.png)\n\n");

                    if self.include_metadata {
                        md.push_str(&format!(
                            "<!-- Figure bbox: [{:.1}, {:.1}, {:.1}, {:.1}] -->\n\n",
                            bbox.x1, bbox.y1, bbox.x2, bbox.y2
                        ));
                    }
                }
                "table" => {
                    if !current_paragraph.is_empty() {
                        md.push_str(&current_paragraph);
                        md.push_str("\n\n");
                        current_paragraph.clear();
                    }

                    md.push_str("| Table |\n");
                    md.push_str("|-------|\n");
                    md.push_str("| [Table content here] |\n\n");
                }
                "equation" => {
                    if !current_paragraph.is_empty() {
                        md.push_str(&current_paragraph);
                        md.push_str("\n\n");
                        current_paragraph.clear();
                    }

                    md.push_str("$$\n[LaTeX equation]\n$$\n\n");
                }
                _ => {}
            }
        }

        // Flush remaining paragraph
        if !current_paragraph.is_empty() {
            md.push_str(&current_paragraph);
            md.push_str("\n\n");
        }

        Ok(md)
    }
}
```

## CLI Enhancement

Update `main.rs` to support new features:

```rust
fn main() -> Result<()> {
    println!("PDF-Mash: PDF to Markdown Converter");

    let args: Vec<String> = std::env::args().collect();
    if args.len() < 2 {
        println!("Usage: pdf-mash <pdf_file> [options]");
        println!("\nOptions:");
        println!("  --model <path>         Model path (default: models/layout.onnx)");
        println!("  --viz                  Save visualization images");
        println!("  --output <file>        Save markdown to file");
        println!("\nExample:");
        println!("  pdf-mash document.pdf --viz --output out.md");
        return Ok(());
    }

    let pdf_path = &args[1];
    let mut model_path = "models/layout.onnx";
    let mut save_viz = false;
    let mut output_file: Option<String> = None;

    // Simple argument parsing
    let mut i = 2;
    while i < args.len() {
        match args[i].as_str() {
            "--model" => {
                if i + 1 < args.len() {
                    model_path = &args[i + 1];
                    i += 2;
                } else {
                    eprintln!("Error: --model requires a path");
                    return Ok(());
                }
            }
            "--viz" => {
                save_viz = true;
                i += 1;
            }
            "--output" => {
                if i + 1 < args.len() {
                    output_file = Some(args[i + 1].clone());
                    i += 2;
                } else {
                    eprintln!("Error: --output requires a filename");
                    return Ok(());
                }
            }
            _ => {
                i += 1;
            }
        }
    }

    println!("Processing: {}", pdf_path);
    println!("Model: {}", model_path);

    let mut processor = Processor::new(model_path, true)?;
    let markdown = processor.process_pdf(pdf_path, save_viz)?;

    if let Some(output_path) = output_file {
        std::fs::write(&output_path, &markdown)?;
        println!("\n✅ Markdown saved to: {}", output_path);
    } else {
        println!("\n=== Generated Markdown ===\n");
        println!("{}", markdown);
    }

    Ok(())
}
```

## Common Issues & Solutions

### Issue 1: NMS Too Aggressive
**Symptom:** Legitimate detections removed
**Solution:** Increase IOU threshold from 0.45 to 0.55

### Issue 2: Reading Order Wrong for Multi-Column
**Symptom:** Right column text appears before left column finishes
**Solution:** Decrease `column_gap_threshold` from 50.0 to 30.0

### Issue 3: Font Not Found Error
**Symptom:** Visualization fails with font loading error
**Solution:** Verify `assets/DejaVuSans.ttf` exists and is valid TTF file

### Issue 4: Memory Usage Increases
**Symptom:** RAM usage grows with visualization enabled
**Solution:** Process pages in batches, clear annotated images after saving
