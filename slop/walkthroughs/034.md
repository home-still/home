### Implement ConfusableMapper

**File:** `pdf-mash/src/text_correction/confusables.rs`

```rust
use std::collections::HashMap;

/// Context-aware character confusion handler for OCR
/// Fixes: 1↔I↔l, 0↔O, rn→m, cl→d, vv→w
pub struct ConfusableMapper {
    bigram_map: HashMap<String, String>,
}

impl ConfusableMapper {
    pub fn new() -> Self {
        let mut bigram_map = HashMap::new();

        // Common OCR bigram errors (visual similarity)
        bigram_map.insert("rn".to_string(), "m".to_string());
        bigram_map.insert("cl".to_string(), "d".to_string());
        bigram_map.insert("vv".to_string(), "w".to_string());
        bigram_map.insert("li".to_string(), "h".to_string());

        Self { bigram_map }
    }

    pub fn deconfuse(&self, text: &str) -> String {
        // First pass: bigram substitution
        let mut result = text.to_string();
        for (pattern, replacement) in &self.bigram_map {
            result = result.replace(pattern, replacement);
        }

        // Second pass: context-aware single character
        let chars: Vec<char> = result.chars().collect();
        let mut fixed = String::with_capacity(result.len());

        for (i, &ch) in chars.iter().enumerate() {
            let prev = if i > 0 { Some(chars[i - 1]) } else { None };
            let next = if i < chars.len() - 1 { Some(chars[i + 1]) } else { None };

            let corrected = match ch {
                '1' if prev.is_some_and(|c| c.is_alphabetic())
                    && next.is_some_and(|c| c.is_alphabetic()) => 'I',
                '0' if prev.is_some_and(|c| c.is_alphabetic())
                    && next.is_some_and(|c| c.is_alphabetic()) => 'O',
                _ => ch,
            };

            fixed.push(corrected);
        }

        fixed
    }
}

mod tests {

    #[test]
    fn test_digit_to_letter_in_word_context() {
        let mapper = ConfusableMapper::new();

        assert_eq!(mapper.deconfuse("1 am gl0rious"), "I am glOrious");
        assert_eq!(mapper.deconfuse("who1have0been"), "whoIhaveObeen");

        // Preserve numbers
        assert_eq!(mapper.deconfuse("I have 10 apples"), "I have 10 apples");
        assert_eq!(mapper.deconfuse("year 2025"), "year 2025");
    }

    #[test]

        let mapper = ConfusableMapper::new();

        assert_eq!(mapper.deconfuse("running"), "muning");
        assert_eq!(mapper.deconfuse("climb"), "dimb");
        assert_eq!(mapper.deconfuse("revving"), "rewiing");
    }
}
```

```rust
mod confusables;

pub use confusables::ConfusableMapper;

use anyhow::Result;
use symspell::{SymSpell, Verbosity};
use unicode_normalization::UnicodeNormalization;

pub struct TextCorrector {
    confusables: ConfusableMapper,
    symspell: SymSpell,
    enabled: bool,
}

#[derive(Debug, Clone)]
pub struct CorrectedText {
    pub text: String,
    pub confidence: f32,
    pub edits_made: usize,
    pub character_fixes: usize,
    pub word_fixes: usize,
}

impl CorrectedText {
    pub fn unchanged(text: &str) -> Self {
        Self {
            text: text.to_string(),
            confidence: 1.0,
            edits_made: 0,
            character_fixes: 0,
            word_fixes: 0,
        }
    }
}

impl TextCorrector {
    pub fn new(dictionary_path: &str, bigram_path: &str) -> Result<Self> {
        println!("Loading text correction dictionaries...");

        let confusables = ConfusableMapper::new();
        let mut symspell = SymSpell::default();

        if std::path::Path::new(dictionary_path).exists() {
            symspell.load_dictionary(dictionary_path, 0, 1, " ")?;
            println!("  ✓ Loaded {} words", symspell.word_count());
        } else {
            anyhow::bail!("Dictionary not found: {}", dictionary_path);
        }

        if std::path::Path::new(bigram_path).exists() {
            symspell.load_bigram_dictionary(bigram_path, 0, 2, " ")?;
            println!("  ✓ Loaded bigram dictionary");
        } else {
            eprintln!("⚠️  Bigram dictionary not found");
        }

        Ok(Self {
            confusables,
            symspell,
            enabled: true,
        })
    }

    pub fn correct(&self, text: &str) -> Result<CorrectedText> {
        if !self.enabled || text.is_empty() {
            return Ok(CorrectedText::unchanged(text));
        }

        // Stage 1: Unicode normalization
        let normalized: String = text.nfc().collect();

        // Stage 2: Character confusion
        let deconfused = self.confusables.deconfuse(&normalized);
        let character_fixes = if deconfused != normalized { 1 } else { 0 };

        // Stage 3: Spelling + segmentation
        let suggestions = self.symspell.lookup_compound(&deconfused, 2);

        let (corrected, confidence, word_fixes) = if let Some(suggestion) = suggestions.first() {
            let edits = suggestion.distance as usize;
            let confidence = if edits == 0 {
                1.0
            } else {
                1.0 - (edits as f32 * 0.1).min(0.5)
            };
            (suggestion.term.clone(), confidence, edits)
        } else {
            (deconfused.clone(), 0.5, 0)
        };

        Ok(CorrectedText {
            text: corrected,
            confidence,
            edits_made: character_fixes + word_fixes,
            character_fixes,
            word_fixes,
        })
    }

    pub fn set_enabled(&mut self, enabled: bool) {
        self.enabled = enabled;
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_full_pipeline() {
        let corrector = TextCorrector::new(
            "models/dictionaries/frequency_dictionary_en_82_765.txt",
            "models/dictionaries/frequency_bigramdictionary_en_243_342.txt",
        ).expect("Failed to load dictionaries");

        let result = corrector.correct("who1have0beentogether").unwrap();
        assert!(result.text.contains("who"));
        assert!(result.text.contains("have"));
        assert!(result.text.contains("been"));
        assert!(result.text.contains("together"));
    }
}
```
