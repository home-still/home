## Mental Model

The codebase has OCR logic scattered across modules with duplication. This refactor:

1. **Consolidates** all OCR into `src/ocr/`
2. **Separates** detection (finding text) from recognition (reading text)
3. **Unifies** text correction (segmentation + confusables + spelling)
4. **Extracts** shared utilities (normalization, session creation)

```
BEFORE:
  models/ocr.rs         → Detection + Recognition + Preprocessing (mixed)
  text_correction/      → Spelling + Confusables (separate)
  ocr/                  → Partial abstractions (incomplete)

AFTER:
  models/ocr.rs         → Detection ONLY (slimmed down)
  ocr/                  → Complete OCR subsystem
    backends/paddle.rs  → Recognition (extracted)
    preprocessor.rs     → Chunking + Normalization (enhanced)
    postprocessor.rs    → Segmentation + Confusables + Spelling (merged)
    pipeline.rs         → Orchestrator (new)
  text_correction/      → DELETED (merged into postprocessor)
```

```rust
use image::GenericImageView;

/// Normalize image to ndarray for ONNX inference.
/// Maps pixel values [0, 255] to [-1, 1].
pub fn normalize_image(image: &DynamicImage) -> ndarray::Array4<f32> {
    let rgb = image.to_rgb8();
    let (width, height) = rgb.dimensions();

    let mut array = ndarray::Array4::<f32>::zeros((1, 3, height as usize, width as usize));

    for y in 0..height {
        for x in 0..width {
            let pixel = rgb.get_pixel(x, y);
            array[[0, 0, y as usize, x as usize]] = (pixel[0] as f32 - 127.5) / 127.5;
            array[[0, 1, y as usize, x as usize]] = (pixel[1] as f32 - 127.5) / 127.5;
            array[[0, 2, y as usize, x as usize]] = (pixel[2] as f32 - 127.5) / 127.5;
        }
    }

    array
}
```

### Step 2.2: Create `ocr/backends/paddle.rs`

Extract ONLY recognition from `models/ocr.rs`:

```rust
use anyhow::{Context, Result};
use image::DynamicImage;
use ort::execution_providers::CUDAExecutionProvider;
use ort::session::{builder::GraphOptimizationLevel, Session};
use tracing::{debug, info};

use crate::ocr::preprocessor::normalize_image;
use crate::ocr::traits::{RecognitionResult, TextRecognizer};

pub struct PaddleRecognizer {
    session: Session,
    char_dict: Vec<String>,
    target_height: u32,
}

impl PaddleRecognizer {
    pub fn new(model_path: &str, dict_path: &str, use_cuda: bool) -> Result<Self> {
        let mut builder = Session::builder()?
            .with_optimization_level(GraphOptimizationLevel::Level3)?;

        if use_cuda {
            debug!("Enabling CUDA for recognition model");
            builder = builder.with_execution_providers([
                CUDAExecutionProvider::default().build()
            ])?;
        }

        let session = builder
            .commit_from_file(model_path)
            .with_context(|| format!("Failed to load recognition model: {}", model_path))?;

        let char_dict = std::fs::read_to_string(dict_path)
            .with_context(|| format!("Failed to load character dictionary: {}", dict_path))?
            .lines()
            .map(String::from)
            .collect();

        info!("PaddleOCR recognizer initialized (CUDA: {})", use_cuda);

        Ok(Self {
            session,
            char_dict,
            target_height: 48,
        })
    }

    fn decode_ctc(&self, shape: &[i64], data: &[f32]) -> (String, f32, Vec<f32>) {
        let sequence_length = shape[1] as usize;
        let num_classes = shape[2] as usize;
        let blank_idx = 0;

        let mut text = String::new();
        let mut char_confidences = Vec::new();
        let mut last_idx = blank_idx;

        for t in 0..sequence_length {
            let mut max_idx = 0;
            let mut max_prob = data[t * num_classes];

            for c in 1..num_classes {
                let prob = data[t * num_classes + c];
                if prob > max_prob {
                    max_prob = prob;
                    max_idx = c;
                }
            }

            if max_idx == blank_idx || max_idx == last_idx {
                last_idx = max_idx;
                continue;
            }

            let char_idx = max_idx - 1;
            if char_idx < self.char_dict.len() {
                text.push_str(&self.char_dict[char_idx]);
                char_confidences.push(max_prob);
            }

            last_idx = max_idx;
        }

        let avg_confidence = if char_confidences.is_empty() {
            0.0
        } else {
            char_confidences.iter().sum::<f32>() / char_confidences.len() as f32
        };

        (text, avg_confidence, char_confidences)
    }
}

impl TextRecognizer for PaddleRecognizer {
    fn recognize(&mut self, image: &DynamicImage) -> Result<RecognitionResult> {
        let input_tensor = normalize_image(image);
        let value = ort::value::Value::from_array(input_tensor)?;

        let outputs = self.session.run(ort::inputs!["x" => value])?;
        let tensor = outputs[0].try_extract_tensor::<f32>()?;
        let shape: Vec<i64> = tensor.0.to_vec();
        let data: Vec<f32> = tensor.1.to_vec();

        let (text, confidence, char_confidences) = self.decode_ctc(&shape, &data);

        Ok(RecognitionResult {
            text,
            confidence,
            char_confidences: Some(char_confidences),
        })
    }

    fn name(&self) -> &str {
        "PaddleOCR-v5"
    }

    fn expected_height(&self) -> u32 {
        self.target_height
    }

    fn warmup(&mut self) -> Result<()> {
        debug!("Warming up PaddleOCR recognizer");
        let dummy = DynamicImage::new_rgb8(100, self.target_height);
        let _ = self.recognize(&dummy)?;
        Ok(())
    }
}
```

```rust
use std::collections::HashMap;
use std::fs::File;
use std::io::Write;
use std::path::Path;

use anyhow::{bail, Result};
use symspell::{AsciiStringStrategy, SymSpell, Verbosity};
use tracing::debug;

use crate::ocr::config::OcrConfig;

#[derive(Debug, Clone)]
pub struct CorrectionEvent {
    pub original: String,
    pub corrected: String,
    pub correction_type: CorrectionType,
}

#[derive(Debug, Clone, Copy)]
pub enum CorrectionType {
    Segmentation,
    Confusable,
    Spelling,
}

pub struct Postprocessor {
    config: OcrConfig,
    dictionary: Option<SymSpell<AsciiStringStrategy>>,
    bigram_map: HashMap<String, String>,
    reverse_map: HashMap<char, String>,
    corrections: Vec<CorrectionEvent>,
}

impl Postprocessor {
    pub fn new(config: OcrConfig, dictionary_path: Option<&str>) -> Result<Self> {
        let needs_dictionary = config.enable_segmentation
            || config.enable_spelling_correction
            || config.enable_confusable_correction;

        let dictionary = match (needs_dictionary, dictionary_path) {
            (true, Some(path)) => {
                if !Path::new(path).exists() {
                    bail!(
                        "Dictionary file not found: '{}'. \
                        Download from: https://github.com/wolfgarbe/SymSpell#frequency-dictionary",
                        path
                    );
                }

                let mut dict: SymSpell<AsciiStringStrategy> = SymSpell::default();
                if !dict.load_dictionary(path, 0, 1, " ") {
                    bail!("Failed to parse dictionary: '{}'", path);
                }

                debug!("Loaded dictionary from {}", path);
                Some(dict)
            }
            (true, None) => {
                bail!(
                    "Text correction enabled but no dictionary path provided. \
                    Either provide a path or disable correction in OcrConfig."
                );
            }
            (false, _) => None,
        };

        // OCR bigram confusions (from Tesseract eng.unicharambigs)
        let mut bigram_map = HashMap::new();
        bigram_map.insert("rn".into(), "m".into());
        bigram_map.insert("cl".into(), "d".into());
        bigram_map.insert("vv".into(), "w".into());
        bigram_map.insert("VV".into(), "W".into());
        bigram_map.insert("li".into(), "h".into());
        bigram_map.insert("lr".into(), "h".into());
        bigram_map.insert("rm".into(), "nn".into());
        bigram_map.insert("ri".into(), "n".into());
        bigram_map.insert("rr".into(), "n".into());

        let mut reverse_map = HashMap::new();
        reverse_map.insert('m', "rn".into());

        Ok(Self {
            config,
            dictionary,
            bigram_map,
            reverse_map,
            corrections: Vec::new(),
        })
    }

    /// Process text through all enabled correction stages.
    /// Order: segmentation → confusables → spelling
    pub fn process(&mut self, text: &str) -> String {
        let mut result = text.to_string();

        if self.config.enable_segmentation {
            result = self.segment_words(&result);
        }

        if self.config.enable_confusable_correction {
            result = self.correct_confusables(&result);
        }

        if self.config.enable_spelling_correction {
            result = self.correct_spelling(&result);
        }

        result
    }

    fn segment_words(&mut self, text: &str) -> String {
        let Some(dict) = &self.dictionary else {
            return text.to_string();
        };

        text.split_whitespace()
            .map(|token| {
                if Self::needs_segmentation(token) {
                    let segmented = dict.word_segmentation(token, 2).segmented_string;
                    if segmented != token {
                        self.corrections.push(CorrectionEvent {
                            original: token.to_string(),
                            corrected: segmented.clone(),
                            correction_type: CorrectionType::Segmentation,
                        });
                    }
                    segmented
                } else {
                    token.to_string()
                }
            })
            .collect::<Vec<_>>()
            .join(" ")
    }

    fn needs_segmentation(token: &str) -> bool {
        token.len() > 15
            && token.chars().all(|c| c.is_alphabetic())
            && !token.contains('-')
    }

    fn correct_confusables(&mut self, text: &str) -> String {
        let Some(dict) = &self.dictionary else {
            return text.to_string();
        };

        let mut result = Vec::new();
        let mut current_word = String::new();

        for ch in text.chars() {
            if ch.is_alphabetic() {
                current_word.push(ch);
            } else {
                if !current_word.is_empty() {
                    let corrected = self.correct_confusable_word(&current_word, dict);
                    result.push(corrected);
                    current_word.clear();
                }
                result.push(ch.to_string());
            }
        }

        if !current_word.is_empty() {
            let corrected = self.correct_confusable_word(&current_word, dict);
            result.push(corrected);
        }

        result.join("")
    }

    fn correct_confusable_word(
        &mut self,
        word: &str,
        dict: &SymSpell<AsciiStringStrategy>,
    ) -> String {
        // Already valid?
        if !dict.lookup(word, Verbosity::Top, 2).is_empty() {
            return word.to_string();
        }

        // Try bigram corrections (rn→m, cl→d, etc.)
        let bigram_corrected = self.apply_bigram_corrections(word);
        if !dict.lookup(&bigram_corrected, Verbosity::Top, 2).is_empty() {
            self.corrections.push(CorrectionEvent {
                original: word.to_string(),
                corrected: bigram_corrected.clone(),
                correction_type: CorrectionType::Confusable,
            });
            return bigram_corrected;
        }

        // Try reverse corrections (m→rn)
        let reverse_corrected = self.apply_reverse_corrections(word);
        if !dict.lookup(&reverse_corrected, Verbosity::Top, 2).is_empty() {
            self.corrections.push(CorrectionEvent {
                original: word.to_string(),
                corrected: reverse_corrected.clone(),
                correction_type: CorrectionType::Confusable,
            });
            return reverse_corrected;
        }

        word.to_string()
    }

    fn apply_bigram_corrections(&self, word: &str) -> String {
        let chars: Vec<char> = word.chars().collect();
        let mut result = String::with_capacity(word.len());
        let mut i = 0;

        while i < chars.len() {
            if i + 1 < chars.len() {
                let bigram: String = [chars[i], chars[i + 1]].iter().collect();
                if let Some(replacement) = self.bigram_map.get(&bigram) {
                    result.push_str(replacement);
                    i += 2;
                    continue;
                }
            }
            result.push(chars[i]);
            i += 1;
        }

        result
    }

    fn apply_reverse_corrections(&self, word: &str) -> String {
        let mut result = String::with_capacity(word.len() * 2);
        for ch in word.chars() {
            if let Some(replacement) = self.reverse_map.get(&ch) {
                result.push_str(replacement);
            } else {
                result.push(ch);
            }
        }
        result
    }

    fn correct_spelling(&mut self, text: &str) -> String {
        let Some(dict) = &self.dictionary else {
            return text.to_string();
        };

        text.split_whitespace()
            .map(|token| {
                if !token.chars().any(|c| c.is_alphabetic()) {
                    return token.to_string();
                }

                if let Some(suggestion) = dict.lookup(token, Verbosity::Top, 2).first() {
                    if suggestion.distance <= 1 && suggestion.term != token {
                        self.corrections.push(CorrectionEvent {
                            original: token.to_string(),
                            corrected: suggestion.term.clone(),
                            correction_type: CorrectionType::Spelling,
                        });
                        return suggestion.term.clone();
                    }
                }

                token.to_string()
            })
            .collect::<Vec<_>>()
            .join(" ")
    }

    pub fn get_corrections(&self) -> &[CorrectionEvent] {
        &self.corrections
    }

    pub fn clear_corrections(&mut self) {
        self.corrections.clear();
    }

    pub fn export_corrections(&self, path: &str) -> std::io::Result<()> {
        let mut file = File::create(path)?;
        writeln!(file, "original,corrected,type")?;

        for event in &self.corrections {
            writeln!(
                file,
                "{},{},{:?}",
                event.original, event.corrected, event.correction_type
            )?;
        }

        Ok(())
    }
}
```

### Step 2.4: Create `ocr/pipeline.rs`

```rust
use anyhow::Result;
use image::{DynamicImage, GenericImageView};
use tracing::{debug, warn};

use crate::ocr::{
    config::OcrConfig,
    postprocessor::{CorrectionEvent, Postprocessor},
    preprocessor::Preprocessor,
    traits::{RecognitionResult, TextRecognizer},
};

pub struct OcrPipeline<R: TextRecognizer> {
    config: OcrConfig,
    preprocessor: Preprocessor,
    recognizer: R,
    postprocessor: Postprocessor,
}

impl<R: TextRecognizer> OcrPipeline<R> {
    pub fn new(
        config: OcrConfig,
        recognizer: R,
        dictionary_path: Option<&str>,
    ) -> Result<Self> {
        let preprocessor = Preprocessor::new(config.clone());
        let postprocessor = Postprocessor::new(config.clone(), dictionary_path)?;

        Ok(Self {
            config,
            preprocessor,
            recognizer,
            postprocessor,
        })
    }

    pub fn extract_text(&mut self, image: &DynamicImage) -> Result<RecognitionResult> {
        // Validate minimum dimensions
        let (width, height) = image.dimensions();
        if width < self.config.min_crop_width || height < self.config.min_crop_height {
            debug!("Image too small ({}x{}), skipping", width, height);
            return Ok(RecognitionResult::empty());
        }

        // Step 1: Chunk if needed
        let chunks = self.preprocessor.prepare(image)?;
        let num_chunks = chunks.len();

        if num_chunks > 1 {
            debug!("Split wide image into {} chunks", num_chunks);
        }

        // Step 2: Recognize each chunk
        let mut chunk_results = Vec::with_capacity(num_chunks);
        for (i, chunk) in chunks.iter().enumerate() {
            let result = self.recognizer.recognize(&chunk.image)?;

            if num_chunks > 1 {
                debug!(
                    "Chunk {}: '{}' (conf: {:.2})",
                    i,
                    truncate(&result.text, 30),
                    result.confidence
                );
            }

            chunk_results.push(result);
        }

        // Step 3: Merge chunks
        let merged = self.merge_chunks(&chunk_results);

        // Step 4: Filter low confidence
        if merged.confidence < self.config.min_confidence {
            warn!(
                "Low confidence {:.2}, discarding: '{}'",
                merged.confidence,
                truncate(&merged.text, 30)
            );
            return Ok(RecognitionResult::empty());
        }

        // Step 5: Post-process
        let corrected_text = self.postprocessor.process(&merged.text);

        Ok(RecognitionResult {
            text: corrected_text,
            confidence: merged.confidence,
            char_confidences: merged.char_confidences,
        })
    }

    fn merge_chunks(&self, results: &[RecognitionResult]) -> RecognitionResult {
        if results.is_empty() {
            return RecognitionResult::empty();
        }

        if results.len() == 1 {
            return results[0].clone();
        }

        // Estimate characters in overlap region (~10 pixels per char at 48px height)
        let overlap_chars = (self.config.chunk_overlap / 10) as usize;

        let mut merged_text = String::new();
        let mut total_confidence = 0.0;

        for (i, result) in results.iter().enumerate() {
            if i == 0 {
                merged_text.push_str(&result.text);
            } else if result.text.len() > overlap_chars {
                merged_text.push_str(&result.text[overlap_chars..]);
            }
            total_confidence += result.confidence;
        }

        RecognitionResult {
            text: merged_text,
            confidence: total_confidence / results.len() as f32,
            char_confidences: None,
        }
    }

    pub fn get_corrections(&self) -> &[CorrectionEvent] {
        self.postprocessor.get_corrections()
    }

    pub fn clear_corrections(&mut self) {
        self.postprocessor.clear_corrections();
    }

    pub fn export_corrections(&self, path: &str) -> std::io::Result<()> {
        self.postprocessor.export_corrections(path)
    }
}

fn truncate(s: &str, max_len: usize) -> String {
    if s.len() <= max_len {
        s.to_string()
    } else {
        format!("{}...", &s[..max_len])
    }
}
```
