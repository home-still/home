```rust
pub fn detect_text_regions(
    &self,
    image: &DynamicImage,
) -> Result<Vec<BBox>, Box<dyn std::error::Error>> {
    // TODO: Get original dimensions
    let original_width = image.width() as f32;
    let original_height = image.height() as f32;

    // TODO: Preprocess to 960x960 with padding
    let (preprocessed, padding_metadata) = preprocess_for_detection(image);

    // TODO: Run detection (returns bboxes in 960x960 space)
    let detections_padded = self.run_detection_model(&preprocessed)?;

    // TODO: Transform each bbox back to original space
    let detections_original = detections_padded.iter().map(|bbox| {
        // Inverse letterbox transformation
        transform_bbox_from_padded(bbox, &padding_metadata, original_width, original_height)
    }).collect();

    Ok(detections_original)
}

fn transform_bbox_from_padded(
    bbox: &BBox,
    padding: &PaddingMetadata,
    orig_width: f32,
    orig_height: f32,
) -> BBox {
    // TODO: Subtract padding offsets
    let x1_unpadded = bbox.x1 - padding.pad_left;
    let y1_unpadded = bbox.y1 - padding.pad_top;
    let x2_unpadded = bbox.x2 - padding.pad_left;
    let y2_unpadded = bbox.y2 - padding.pad_top;

    // TODO: Scale back to original dimensions
    let scale_x = orig_width / padding.preprocessed_width;
    let scale_y = orig_height / padding.preprocessed_height;

    BBox {
        x1: x1_unpadded * scale_x,
        y1: y1_unpadded * scale_y,
        x2: x2_unpadded * scale_x,
        y2: y2_unpadded * scale_y,
        ..bbox.clone()
    }
}
```

### Step 7: Markdown Table Generation

**Pattern Recognition:**

Convert cell grid to markdown table syntax:

```markdown
| Header 1 | Header 2 | Header 3 |
|----------|----------|----------|
| Cell 1   | Cell 2   | Cell 3   |
```

**Let's code:**

```rust
// pdf-mash/src/pipeline/markdown_generator.rs

impl MarkdownGenerator {
    pub fn generate_table(
        &self,
        cells: &[Cell],
    ) -> String {
        // TODO: Find max row and col
        let max_row = cells.iter().map(|c| c.row).max().unwrap_or(0);
        let max_col = cells.iter().map(|c| c.col).max().unwrap_or(0);

        // TODO: Create 2D grid
        let mut grid: Vec<Vec<Option<String>>> = vec![
            vec![None; max_col + 1]; max_row + 1
        ];

        // TODO: Fill grid with cell text
        for cell in cells {
            if let Some(text) = &cell.text {
                grid[cell.row][cell.col] = Some(text.clone());
            }
        }

        // TODO: Generate markdown
        let mut markdown = String::new();

        // Header row
        markdown.push_str("|");
        for col in 0..=max_col {
            let text = grid[0][col].as_deref().unwrap_or("");
            markdown.push_str(&format!(" {} |", text));
        }
        markdown.push('\n');

        // Separator
        markdown.push_str("|");
        for _ in 0..=max_col {
            markdown.push_str("----------|");
        }
        markdown.push('\n');

        // Data rows
        for row in 1..=max_row {
            markdown.push_str("|");
            for col in 0..=max_col {
                let text = grid[row][col].as_deref().unwrap_or("");
                markdown.push_str(&format!(" {} |", text));
            }
            markdown.push('\n');
        }

        markdown
    }
}
```

```rust
        match formula.formula_type {
            FormulaType::Inline => {
                self.output.push_str(&format!(" ${}$ ", latex));
            }
            FormulaType::Block => {
                self.output.push_str(&format!("\n$$\n{}\n$$\n", latex));
            }
        }
```
